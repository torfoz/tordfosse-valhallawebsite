# Klasse for 친 kj칮re roboten fremover, bakover, rotere og rette opp roboten

            self.robot.drive(100, 0)

    # Konstrukt칮r for klassen BuildBridge
    # Tar inn retning som parameter og setter antall rotasjoner motoren skal gj칮re
